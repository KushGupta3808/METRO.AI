STREAMING_CHUNK: Setting up clean single-stage Python 3.12 environment

FROM python:3.12-slim

WORKDIR /app

STREAMING_CHUNK: Installing compiler and Postgres client dependencies

RUN apt-get update && apt-get install -y --no-install-recommends 

build-essential 

libpq-dev 

gcc 

&& rm -rf /var/lib/apt/lists/*

STREAMING_CHUNK: Installing Python requirements globally

COPY requirements.txt .
RUN pip install --no-cache-dir -r requirements.txt

STREAMING_CHUNK: Copying application code files

COPY . .

STREAMING_CHUNK: Configuring environment variables

ENV PYTHONUNBUFFERED=1
ENV PYTHONDONTWRITEBYTECODE=1
ENV METRO_AI_ENV=production

EXPOSE 8000

STREAMING_CHUNK: Starting the Uvicorn web server

CMD ["uvicorn", "main:app", "--host", "0.0.0.0", "--port", "8000"]
